1 Install gulp-postcss and postcss-preset-env 
npm install gulp-postcss --save-dev
 npm install postcss-preset-env --save-dev
2 Use PostCSS Preset Env in your Gulpfile: 
import postcss from 'gulp-postcss';
 import postcssPresetEnv from 'postcss-preset-env';		
 postcss([
	 postcssPresetEnv(/* options */)
 ]).process(YOUR_CSS);
3 Create postcss.config.js and add PostCSS Preset Env 
{
   "postcss": {
	 "parser": "sugarss",
	 "map": false,
	 "from": "/path/to/src.sss",
	 "to": "/path/to/dest.css",
	 "plugins": {
	   "postcss-preset-env": {}
	 }
   }
 }
